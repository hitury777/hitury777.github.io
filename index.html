function redirectBasedOnPlatform() { const userAgent = window.navigator.userAgent || window.navigator.vendor || window.opera; // 檢測是否是 iOS 平台 (iPhone, iPad, iPod) if (/iPhone|iPad|iPod/i.test(userAgent)) { // 跳轉到 Apple Store 的應用下載頁面 window.location.href = "https://apps.apple.com/tw/app/%E7%BE%A4%E7%9B%8A%E8%A1%8C%E5%8B%95%E8%B4%8F%E5%AE%B6/id1469014227"; } // 檢測是否是 Android 平台 else if (/Android/i.test(userAgent)) { // 跳轉到 Google Play 的應用下載頁面 window.location.href = "https://play.google.com/store/apps/details?id=com.SK.HappyTrader"; } // 其他平台或桌面 else { // 跳轉到網站或其他應用的下載頁面 window.location.href = "https://www.google.com.tw/?hl=zh_TW"; } } // 當頁面加載時自動跳轉 window.onload = redirectBasedOnPlatform;
